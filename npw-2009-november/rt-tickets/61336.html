if split() returns fewer elements than the list assignment expects (a
valid condition), it crashes when trying to assign from an element that
doesn't exist.

# WORKS
{
my $handler = 'Foo::Bar.baz';
my ($class, $method) = split('.', $handler);
say "class=$class";
say "method=$method";
}

# FAILS
{
my $handler = 'Foo::Bar';
my ($class, $method) = split('.', $handler);
say "class=$class";
say "method=$method";
}

# OUTPUT

method=baz
