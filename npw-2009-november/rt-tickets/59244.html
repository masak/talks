-.sub 'infix:cmp'
+.sub 'infix:_cmp' :multi(_,_)
.param pmc a
.param pmc b
$I0 = cmp a, b
diff --git a/languages/perl6/src/classes/Pair.pir b/languages/perl6/src/classes/Pair.pir
index 2aa69b0..cf697b7 100644
--- a/languages/perl6/src/classes/Pair.pir
+++ b/languages/perl6/src/classes/Pair.pir
@@ -93,6 +93,33 @@ Returns a Perl code representation of the pair.
.return $P0.'new'('key'=>key, 'value'=>value)
.end

+=item C<infix:cmp>
+
+Compare 2 pairs. Returns true if both keys and values are equals.
+
+=cut
+
+.sub 'infix:cmp' :multi('Pair','Pair')
+ .param pmc a
+ .param pmc b
+
+ say 'Foo'
+
+ .local pmc first_key
+ .local pmc second_key
+ first_key = a.'key'()
+ second_key = b.'key'()
+ $I0 = 'infix:cmp'(first_key, second_key)
+ unless $I0 goto not_equal
+
+ .local pmc first_value
+ .local pmc second_value
+ first_value = a.'value'()
+ second_value = b.'value'()
+ .return 'infix:cmp'(first_value, second_value)
+ not_equal:
+ .return ($I0)
+.end

=back

method can't seem to handle comparison functions that have :multi
