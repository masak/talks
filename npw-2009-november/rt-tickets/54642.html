+.sub 'map' :method
+ .param pmc expression
+ .local pmc res, elem, mapres, newitem
+ .local int len, i
+
+ res = new 'List'
+ len = elements self
+ i = 0
+ loop:
+ if i == len goto done
+ elem = self[i]
+ inc i
+ mapres = expression(elem)
+
+ # flatten mapres
+ mapres = 'list'(mapres)
+ copy_loop:
+ unless mapres goto loop
+ newitem = shift mapres
+ res.'push'(newitem)
+ goto copy_loop
+
+
+ done:
+
+ .return(res)
+.end
=back
