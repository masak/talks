Use of subset defaults to a defined (but false) value when it should
probably be undefined (or should it?).

my subset Strand where {$_ == 1 | 0 | -1};
say Strand.defined; # 1
my Strand $x;
say $x; # Any

my subset Strand2 of Int where {$_ == 1 | 0 | -1};
say Strand2.defined; # 1
my Strand2 $x2;
say $x2; # 0

my subset Strand3 of Str where {$_ eq 1 | 0 | -1};
say Strand3.defined; # 1
my Strand3 $x3;
say $x3; # ""
