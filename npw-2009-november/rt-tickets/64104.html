token TOP { ^ <value> $ {*} }
token value { \d+ {*} }
}
class Actions {
method value($/) { say 'in value()'; make 1..$/};
method TOP($/) { say 'in TOP()'; make 1 + $/<value>};
}

say $x ?? 'matched' !! 'no match';
