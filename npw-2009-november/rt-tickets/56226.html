+.sub 'roots' :method :multi(_)
+ .param int n
+ .local num x, mag, angle, abs, twoPI
+ .local pmc retv
+ N1 = pi()
+ N1 *= 2
+ twoPI = N1
+ x = self
+ P1 = 'polar'(x)
+ mag = P1[0]
+ angle = P1[1]
+ unless angle < 0 goto skip
+ angle += twoPI
+ skip:
+ $N0 = 1 / n
+ abs = mag ** $N0
+ P1 = new 'List'
+ retv = P1
+ I1 = 0
+ loop:
+ N1 = twoPI * I1
+ N1 += angle
+ N1 /= n
+ P1 = 'unpolar'(abs, N1)
+ retv.push(P1)
+ I1 += 1
+ if I1 < n goto loop
+ done:
+ .return (retv)
+.end
+
=item rand()

=cut
